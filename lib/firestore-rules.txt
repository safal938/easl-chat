// Firestore Security Rules for Drug Safety Cache
// Add these rules to your Firestore security rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Drug Safety Cache Collection Rules
    match /drugSafetyCache/{cacheId} {
      // Users can only read/write their own cached analyses
      allow read, write: if request.auth != null 
        && request.auth.uid == resource.data.userId;
      
      // Ensure session isolation on create/update
      allow create: if request.auth != null 
        && request.auth.uid == request.resource.data.userId
        && request.resource.data.chatId is string
        && request.resource.data.messageId is string
        && request.resource.data.contentHash is string;
      
      // Prevent cross-user data access
      allow update: if request.auth != null 
        && request.auth.uid == resource.data.userId
        && request.auth.uid == request.resource.data.userId;
    }
    
    // Existing user chat rules (maintain existing functionality)
    match /users/{userId}/chats/{chatId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      match /messages/{messageId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
  }
}
